277  result.at(0) = INT16_MAX;

279  result.at(0) = INT16_MIN;
280  else
281  result.at(0) = static_cast<int16_t>(norm.x * static_cast<float>(INT16_MAX));
282 
283  //now y

287  result.at(1) = INT16_MIN;

327  uint32_t package(const glm::vec4& colour)
328  {
329  //int to return the result, initialised to 0.
330  uint32_t result = 0;
331 
332  //vars for each of the colour channels; R, G, B & A.
333  //converting them from 0-1 to 0-255, static casting from float to uint32_T
334  uint32_t r = (static_cast<uint32_t>(colour.r * 255.0f)) << 0; //bytewise - 000R
335  uint32_t g = (static_cast<uint32_t>(colour.g * 255.0f)) << 8; //00B0
336  uint32_t b = (static_cast<uint32_t>(colour.b * 255.0f)) << 16; //0G00
337  uint32_t a = (static_cast<uint32_t>(colour.a * 255.0f)) << 24; //A000
338 
339  result = (r | g | b | a);
340  return result;
341  }
