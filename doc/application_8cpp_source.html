226  std::array<int16_t, 3> normalise(const glm::vec3& norm)
227  {
228  std::array<int16_t, 3> result;
229 
230  //do x first.
231  if (norm.x == 1.0)
232  result.at(0) = INT16_MAX;
233  else if (norm.x == -1.0)
234  result.at(0) = INT16_MIN;
235  else
236  result.at(0) = static_cast<int16_t>(norm.x * static_cast<float>(INT16_MAX));
237 
238  //now y
239  if (norm.y == 1.0)
240  result.at(1) = INT16_MAX;
241  else if (norm.y == -1.0)
242  result.at(1) = INT16_MIN;
243  else
244  result.at(1) = static_cast<int16_t>(norm.y * static_cast<float>(INT16_MAX));
245 
246  //and now z
247  if (norm.z == 1.0)
248  result.at(2) = INT16_MAX;
249  else if (norm.z == -1.0)
250  result.at(2) = INT16_MIN;
251  else
252  result.at(2) = static_cast<int16_t>(norm.z * static_cast<float>(INT16_MAX));
253 
254  //finally, return the result.
255  return result;
256  };

282  uint32_t package(const glm::vec4& colour)
283  {
284  //int to return the result, initialised to 0.
285  uint32_t result = 0;
286 
287  //vars for each of the colour channels; R, G, B & A.
288  //converting them from 0-1 to 0-255, static casting from float to uint32_T
289  uint32_t r = (static_cast<uint32_t>(colour.r * 255.0f)) << 0; //bytewise - 000R
290  uint32_t g = (static_cast<uint32_t>(colour.g * 255.0f)) << 8; //00B0
291  uint32_t b = (static_cast<uint32_t>(colour.b * 255.0f)) << 16; //0G00
292  uint32_t a = (static_cast<uint32_t>(colour.a * 255.0f)) << 24; //A000
293 
294  result = (r | g | b | a);
295  return result;
296  }
