226  void Renderer2D::RtoRGBA(unsigned char * rBuffer, uint32_t width, uint32_t height)
227  {
228  memset(s_data->glyphBuffer.get(), 0, s_data->glyphBufferSize);
229 
230  //ptr walker pointing towards the result.
231  unsigned char * pWalker = s_data->glyphBuffer.get();
232 
233  for (int32_t i = 0; i < height; i++)
234  {
235  for (int32_t j = 0; j < width; j++)
236  {
